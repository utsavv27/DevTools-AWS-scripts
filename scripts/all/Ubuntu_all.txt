Memory update

#!/bin/bash

# Set Swap File Size (Change as needed)
SWAP_SIZE="8G"
SWAP_FILE="/swapfile"

echo "üöÄ Increasing Swap Memory to $SWAP_SIZE..."

# Step 1: Disable existing swap if any
echo "üîÑ Disabling existing swap..."
sudo swapoff -a

# Step 2: Create a new swap file
echo "üìÅ Creating swap file at $SWAP_FILE..."
sudo fallocate -l $SWAP_SIZE $SWAP_FILE || sudo dd if=/dev/zero of=$SWAP_FILE bs=1M count=$((8 * 1024))

# Step 3: Set correct permissions
echo "üîí Setting permissions..."
sudo chmod 600 $SWAP_FILE

# Step 4: Format as swap space
echo "‚ö° Formatting swap file..."
sudo mkswap $SWAP_FILE

# Step 5: Enable swap
echo "‚úÖ Enabling swap..."
sudo swapon $SWAP_FILE

# Step 6: Make swap permanent
echo "üìù Adding swap entry to /etc/fstab..."
echo "$SWAP_FILE none swap sw 0 0" | sudo tee -a /etc/fstab

# Step 7: Verify
echo "üìä Verifying swap..."
free -h
swapon --show

echo "üöÄ Swap memory increased successfully!"

sudo service apache2 restart

# extra

#   sudo service apache2 restart

## configuration of phpmyadmin config
sudo apt-get install phpmyadmin

    Add the following configuration to this file:
        # phpMyAdmin configuration
        Alias /phpmyadmin /usr/share/phpmyadmin

        <Directory /usr/share/phpmyadmin>
            AddDefaultCharset UTF-8
            <IfModule mod_authz_core.c>
                # Apache 2.4
                Require all granted
            </IfModule>
            <IfModule !mod_authz_core.c>
                # Apache 2.2
                Order Allow,Deny
                Allow from all
            </IfModule>
        </Directory>

        <Directory /usr/share/phpmyadmin/setup/>
            <IfModule mod_authz_core.c>
                # Apache 2.4

            </IfModule>

Redis

sudo apt-get install lsb-release curl gpg
curl -fsSL https://packages.redis.io/gpg | sudo gpg --dearmor -o /usr/share/keyrings/redis-archive-keyring.gpg
sudo chmod 644 /usr/share/keyrings/redis-archive-keyring.gpg
echo "deb [signed-by=/usr/share/keyrings/redis-archive-keyring.gpg] https://packages.redis.io/deb $(lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/redis.list
sudo apt-get update
sudo apt-get install redis

sudo systemctl enable redis-server
sudo systemctl start redis-server
